n_samp=250

sim_df_const = 
  tibble(
    x = rnorm(n_samp, 1, 1),
    error = rnorm(n_samp, 0, 1),
    y = 2 + 3 * x + error
  )

sim_df_nonconst = sim_df_const |> 
  mutate(
  error = error * .75 * x,
  y = 2 + 3 * x + error
)

sim_df_nonconst |>
  ggplot(aes(x=x,y=y))+
  geom_point()

Draw and analyze a bootstrap sample

start with a lil function

boot_sample = function(df) {
  sample_frac(df, replace = TRUE)
}

see how this work

boot_sample(sim_df_nonconst) |>
